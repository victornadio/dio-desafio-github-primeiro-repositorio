CONTROLE DE VERSÃO LOCAL
git add .
OU
git add -A

COMMIT
git commit -m "Inclusão das anotações do curso de Git/GitHub"

ADICIONAR A ORIGEM
git remote add origin (link copiado no GitHub)

DAR PUSH
git push origin main
OU
git push origin master

CONFIGURAR USUARIO NO GITBASH
git config --global user.name "SEU USER"

CONFIGURAR E-MAIL NO GITBASH
git config --global user.email "SEU E-MAIL"

INICIAR A SUBIR
git init

VER SE O GIT IDENTIFICOU
git status
